### state

- ReactJS 어플 내에서 데이터를 보관 → 자동 리렌더링

```JS
  let data = React.useState();

  console.log(data)

---------------------------------------------------------------------
  > (2)[undefined,f]

    : 배열의 첫 번째 값은 데이터(초기값)
    : 두 번째는 함수로써 첫 번째 데이터를 변화시키는 함수이다.


  let data = React.useState(0);

  console.log(data)
  ---------------------------------------------------------------------
  > (2)[0,f]
  : useState()의 기본값 0

```

```JS

  const root = document.getElementById('root');

  // state의 배열 값을 가져오는 방식 1
  function App() {
    const data = React.useState(0);
    return (
      <div>
        <h3>Total clicks : {data[0]}</h3>
        <button>Click me</button>
      </div>
    )
  }

  // state의 배열 값을 가져오는 방식 2
  function App() {
    const data = React.useState(0);
    const counter = data[0];
    const modifier = data[1];
    return (
      <div>
        <h3>Total clicks : {counter}</h3>
        <button>Click me</button>
      </div>
    )
  }

  // state의 배열 값을 가져오는 방식 3
  function App() {
    const [counter, modifier] = React.useState(0);
    return (
      <div>
        <h3>Total clicks : {counter}</h3>
        <button>Click me</button>
      </div>
    )
  }


  ReactDOM.render(<App />, root);
```

### React.useState()

```HTML
<!DOCTYPE html>
<html lang="en">

<body>
  <div id="root"></div>
</body>
<script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
<script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
<script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
<script type="text/babel">

  const root = document.getElementById('root');

  function App() {
    // state를 변경할 때, App() Component는 새로운 값을 가지고 다시 한 번 렌더링 → UI refresh
    const [counter, setCounter] = React.useState(0);

    // 버튼 클릭 → setCounter 함수 호출 → 새 counter 값으로 counter 호출
    const onClick = () => {
      // setCounter(counter + 1);
      // 위의 코드와 같지만 아래쪽이 더 안전하다.
      setCounter((current) => current + 1);
    }

    console.log("rendered");
    console.log(counter);

    return (
      <div>
        <h3>Total clicks : {counter}</h3>
        <button onClick={onClick}>Click me</button>
      </div>
    )
  }

  ReactDOM.render(<App />, root);

</script>

</html>
```

### Unit Conversion (단위 변환)

```HTML
<!DOCTYPE html>
<html lang="en">

<body>
  <div id="root"></div>
</body>
<script crossorigin src="https://unpkg.com/react@18/umd/react.development.js"></script>
<script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
<script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
<script type="text/babel">

  const root = document.getElementById('root');

  // React development CDN을 사용하면 <lable for="">이 아닌 <label htmlFor="">, <h1 class="">가 아닌 <h1 className=""> 속성을 사용해야 한다.
  function App() {

    let [amount, setAmount] = React.useState(0);
    let [inverted, setInverted] = React.useState(false);

    const onChange = (event) => {
      setAmount(event.target.value)
    }
    const reset = () => setAmount(0);
    const onInvert = () => {
      reset();
      setInverted((current) => !current)
    }

    return (
      <div>
        <h1 className="hi">Super Converter</h1>
        <div>
          <label htmlFor="minutes"> Minutes </label>
          <input id="minutes"
            type="number"
            placeholder="Minutes"
            value={inverted ? Math.round(amount * 60) : amount}
            onChange={onChange}
            disabled={inverted}
          />
        </div>
        <div>
          <label htmlFor="hours"> Hours </label>
          <input id="hours"
            type="number"
            placeholder="Hours"
            value={!inverted ? Math.round(amount / 60) : amount}
            onChange={onChange}
            disabled={!inverted}
          />
        </div>
        <button onClick={reset}>Reset</button>
        <button onClick={onInvert}>{inverted ? "Turn back" : "Invert"}</button>
      </div>
    )
  }

  ReactDOM.render(<App />, root);

</script>

</html>
```
