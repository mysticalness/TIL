### Component Life Cycle

```js
import React from "react";


// App class는 React Component를 확장한 Component. 때문에 render method가 존재.
// Function Component : 뭔가를 return → screen에 표시
// class component : react component로 부터 확장 → screen에 표시
// react는 자동적으로 class component의 render method를 실행한다.
class App extends React.Component {
  // 1. component가 mount 될 때, component가 screen에 표시 될 때, component가 website에 갈 때 , constructor를 호출
  constructor(props) {
    super(props);
    console.log("hello");
  }

  state = {
    count: 0,
  };

  add = () => {
    this.setState((current) => ({ count: current.count + 1 }));
  };

  minus = () => {
    this.setState((current) => ({ count: current.count - 1 }));
  };

  // 3. component가 render할 때,
  componentDidMount() {
    console.log("component rendered");
  }

  // add나 minus 버튼을 눌러서 업데이트 될 때
  componentDidUpdate() {
    console.log("I just updated!");
  }

  render() {
    // 2. render
    console.log("I'm rendering");
    const { isLoading } = this.state;
    return (
      <div>
        <h1>The number is : {this.state.count}</h1>
        <button onClick={this.add}>Add</button>
        <button onClick={this.minus}>Minus</button>
      </div>
    );
  }
}

export default App;
```
